% --- Slide 2: DreamZero Recap ---
\begin{frame}{DreamZero Recap: WAMs for Zero-shot Policies}
\hi{World Action Models (WAMs):}
\begin{itemize}
    \item \hi{Joint} video + action generation
          from a single DiT backbone
    \item Autoregressive chunk-wise inference
          with ground-truth injection
    \item \positive{2$\times$} over VLAs,
          \positive{38$\times$} speedup to real-time 7Hz
\end{itemize}

\vspace{0.5em}
\begin{keybox}[Key Paradigm]
WAMs generate ``\hi{how to move}'' by predicting
``\hi{what happens next}'' ---
inverse dynamics emerge from video understanding.
\end{keybox}
\end{frame}

% --- Slide 2b: From DreamZero to DreamDojo ---
\begin{frame}{From DreamZero to DreamDojo}
\begin{columns}[T]
\column{0.48\textwidth}
\hi{DreamZero (WAM = Policy):}
\begin{itemize}
    \item Joint video + action prediction
    \item \hi{Inverse} dynamics:
          video $\rightarrow$ action
    \item \positive{Zero-shot policies}
    \item Trained on robot data
\end{itemize}

\column{0.48\textwidth}
\hi{DreamDojo (World Model):}
\begin{itemize}
    \item Action-conditioned video prediction
    \item \hi{Forward} dynamics:
          action $\rightarrow$ video
    \item \positive{Open-world simulation}
    \item Pretrained on \higold{human video}
\end{itemize}
\end{columns}

\vspace{0.5em}
\begin{keybox}[The Shift]
Same team, complementary paradigms:
DreamZero is a \hi{policy}, DreamDojo is a \hi{simulator}.
\end{keybox}
\end{frame}

% --- Slide 3: The Data Gap ---
\begin{frame}{The Data Gap: Robot Data Is Not Enough}
\hi{The fundamental bottleneck:}
\begin{itemize}
    \item Existing robot datasets: limited coverage,
          narrow distributions
    \item DROID: 350 hours, 86 skills, 564 scenes
    \item AgiBot World: 2.9K hours, 87 skills, 106 scenes
    \item World models trained on robot data
          $\rightarrow$ \negative{in-distribution only}
\end{itemize}

\vspace{0.5em}
\begin{highlightbox}[The Core Problem]
Robot world models plateau at \negative{observed setups}
and are \negative{unresponsive to counterfactual actions}
--- actions the robot never performed in training
(e.g., patting a toy, reaching and missing).
Scaling teleoperation alone cannot cover the long tail.
\end{highlightbox}
\end{frame}

% --- Slide 8: The Action Label Problem ---
\begin{frame}{The Action Label Problem}
\hi{Human videos are vast but lack action labels.}

\vspace{0.5em}
\begin{itemize}
    \item Passive video prediction ignores \negative{causality}
          between observations and actions
    \item Off-the-shelf hand pose extractors (HaMeR)
          miss arm movements, locomotion,
          and fail under occlusion
    \item Converting heterogeneous action formats into
          a unified representation
          $\rightarrow$ significant engineering burden
\end{itemize}

\vspace{0.5em}
\begin{highlightbox}[The Dilemma]
Actionless pretraining wastes interaction knowledge.
Ground-truth actions require expensive capture devices.
We need a \hi{self-supervised} middle ground.
\end{highlightbox}
\end{frame}

% --- Slide 9: LAM Architecture ---
\begin{frame}{Latent Action Model (LAM)}
\begin{center}
\includegraphics[width=0.88\textwidth,
  height=0.38\textheight,keepaspectratio]{%
  ../Figures/DreamDojo/lam_compressed.pdf}
\end{center}

\vspace{0.2em}
\begin{eqbox}
\hi{VAE with information bottleneck:}
\begin{equation*}
\mathcal{L}^{pred}_{\theta,\phi}(f^{t+1})
  = \mathbb{E}_{q_{\phi}(\hat{a}|f^{t:t+1})}
    \log p_{\theta}(f^{t+1}|\hat{a},f^{t})
  - \beta\,D_{KL}(q_{\phi}(\hat{a}|f^{t:t+1})||p(\hat{a}))
\end{equation*}
{\small $\hat{a} \in \mathbb{R}^{32}$, \ \ $\beta=10^{-6}$
(tight bottleneck $\rightarrow$ forces action disentanglement)}
\end{eqbox}
\end{frame}

% --- Slide 10: Latent Actions Cross Embodiments ---
\begin{frame}{Latent Actions Cross Embodiments}
\begin{keybox}[Cross-Embodiment Transfer]
The same latent action $\hat{a}$ captures the
\hi{same physical interaction} regardless of whether
it is performed by a human hand, a gripper,
or a dexterous robot.
\end{keybox}

\vspace{0.5em}
\hi{LAM Architecture (700M params):}
\begin{itemize}
    \item Spatiotemporal Transformer
          \citep{bruce2024genie}
    \item 24 encoder blocks $+$ 24 decoder blocks
    \item Encoder: $(f^t, f^{t+1}) \rightarrow \hat{a}$
          \ \ Decoder: $(\hat{a}, f^t) \rightarrow \hat{f}^{t+1}$
    \item Trained on mixture of \hi{all} datasets
          (human + robot)
\end{itemize}

\end{frame}

% --- Slide 12: Cosmos-Predict2.5 Backbone ---
\begin{frame}{Cosmos-Predict2.5 Backbone}
\begin{methodbox}[Foundation]
\begin{itemize}
    \item Latent video diffusion model
          \citep{ali2025world}
    \item \hi{WAN 2.2 tokenizer}:
          $4\times$ temporal compression\\
          {\small (video latent $x^i$ corresponds to
          4 pixel frames $f^{i:i+4}$)}
    \item \hi{DiT} blocks \citep{peebles2023scalable}
          with cross-attention for text,
          AdaLN for timestep
    \item Trained with \hi{flow matching} loss
    \item Two variants: \higold{2B} and \higold{14B} parameters
\end{itemize}
\end{methodbox}

\vspace{0.3em}
\hi{Training setup:}
\begin{itemize}
    \item 140K steps pretrain (256 H100s),
          50K steps post-train (128 H100s)
    \item Resolution: $640 \times 480$,
          sequences of 13 latent frames
    \item EMA maintained throughout
\end{itemize}
\end{frame}

% --- Slide 12b: Flow Matching in 60 Seconds ---
\begin{frame}{Flow Matching in 60 Seconds}
\begin{keybox}[Intuition]
Learn a \hi{velocity field} that transports
noise $\rightarrow$ data along straight paths.
Simpler and faster than standard diffusion.
\end{keybox}

\vspace{0.5em}
\begin{eqbox}
\hi{Flow matching loss:}
\begin{equation*}
\mathcal{L}_{\text{flow}}(\theta)
  = \mathbb{E}_{\mathbf{x},\epsilon,\mathbf{c},t}
    \left\|\mathbf{u}(\mathbf{x}_{t},t,\mathbf{c};\theta)
    - \mathbf{v}_{t}\right\|^2
\end{equation*}
{\small where $\mathbf{v}_t = \epsilon - \mathbf{x}$
(velocity = noise $-$ clean data)}
\end{eqbox}

\vspace{0.5em}
\begin{itemize}
    \item $t=0$: pure noise.
          $t=1$: clean data.
    \item Model predicts the \hi{velocity} pointing
          from noise toward data
    \item Conditions $\mathbf{c}$: text, frames,
          and \higold{actions} (for world models)
\end{itemize}
\end{frame}

% --- Slide 13: Two Key Innovations ---
\begin{frame}{Two Key Innovations for Controllability}
\begin{center}
\includegraphics[width=0.80\textwidth,
  height=0.35\textheight,keepaspectratio]{%
  ../Figures/DreamDojo/design_compressed.pdf}
\end{center}

\begin{methodbox}[Design Choices]
\begin{enumerate}
    \item \hi{Relative actions:} Rebaseline every 4 timesteps\\
          {\small Concentrates action space
          $\rightarrow$ better generalization to compositions}
    \item \hi{Chunked injection:} 4 consecutive actions
          $a^{t:t+4}$ sent to matching latent frame\\
          {\small Respects causality
          $\rightarrow$ future actions do not leak}
\end{enumerate}
\end{methodbox}
\end{frame}

% --- Slide 14: Action Embedding Equation ---
\begin{frame}{Action Embedding: AdaLN Injection}
\begin{eqbox}
\hi{Action injection pathway:}
\begin{equation*}
\text{actions} \xrightarrow{\text{MLP}}
  \text{action embedding}
  \xrightarrow{+}
  \text{timestep embedding}
  \xrightarrow{\text{AdaLN}}
  \text{DiT blocks}
\end{equation*}
{\small Scale, shift, and gate modulations
in every DiT block}
\end{eqbox}

\vspace{0.5em}
\hi{Key implementation details:}
\begin{itemize}
    \item Last layer of action MLP
          \hi{zero-initialized}
          \citep{zhang2023adding}\\
          {\small Avoids perturbing pretrained state
          at start of training}
    \item During post-training: \hi{reset first layer only},
          finetune all weights
    \item Classifier-free guidance \hi{disabled}
          (empirically no benefit)
\end{itemize}
\end{frame}

% --- Slide 14b: Action Embedding Code Pattern ---
\begin{frame}{Action Embedding: Code Insights}
\begin{methodbox}[Implementation Pattern]
\begin{itemize}
    \item Latent actions projected via lightweight MLP
          to match timestep embedding dim
    \item \hi{Zero-init} on last MLP layer
          $\rightarrow$ identity-like at init
    \item Added to timestep embeddings before AdaLN
    \item Dual pathway: actions modulate through
          \hi{both} timestep AND scale/shift/gate
\end{itemize}
\end{methodbox}

\vspace{0.5em}
\begin{softbox}
\higold{My Take:} The dual-path action injection is more
sophisticated than the paper describes --- actions
modulate DiT through \hi{both timestep AND AdaLN}
scale/shift/gate pathways. The zero-init trick from
ControlNet is crucial for preserving pretrained physics.
\end{softbox}
\end{frame}

% --- Slide 15: Temporal Consistency Loss ---
\begin{frame}{Temporal Consistency Loss}
\hi{Problem:} Flow matching supervises frames individually,
ignoring temporal correlations.

\vspace{0.3em}
\begin{eqbox}
\hi{Temporal consistency loss:}
\begin{equation*}
\mathcal{L}_{\text{temporal}}(\theta)
  = \mathbb{E}\Big[\sum^{K-1}_{i=1}
    \left\|(z^{i+1}-z^{i})-(v^{i+1}-v^{i})\right\|^2\Big]
\end{equation*}
\end{eqbox}

\vspace{0.2em}
\begin{eqbox}
\hi{Final objective:}
$\mathcal{L}_{\text{final}}
  = \mathcal{L}_{\text{flow}}
  + \lambda\,\mathcal{L}_{\text{temporal}},
  \ \ \lambda = 0.1$
\end{eqbox}

\end{frame}

% --- Slide 18: The Distillation Challenge ---
\begin{frame}{The Distillation Challenge}
\hi{Why distill?}
\begin{itemize}
    \item Teacher: \negative{35 denoising steps},
          bidirectional attention, 2.72 FPS
    \item Too slow for live teleoperation
          and online planning
\end{itemize}

\vspace{0.5em}
\begin{highlightbox}[Two Conversions Required]
\begin{enumerate}
    \item \hi{Bidirectional $\rightarrow$ Autoregressive}
          (causal attention, streaming)
    \item \hi{35 steps $\rightarrow$ 4 steps}
          (few-step generation)
\end{enumerate}
\end{highlightbox}

\vspace{0.3em}
Student initialized from teacher weights,
with bidirectional attention replaced by
\hi{causal attention} over a sliding window of 12 frames.
\end{frame}

% --- Slide 19a: Self Forcing Warmup ---
\begin{frame}{Self Forcing: Warmup Stage}
\hi{Stage 1: Warmup} --- student regresses to teacher ODE solutions

\vspace{0.5em}
\begin{eqbox}
\begin{equation*}
\mathcal{L}_{\text{warmup}}
  = \mathbb{E}_{x,t}\|G_{\text{student}}(x_{t},t)-x_{0}\|^2
\end{equation*}
{\small Student generates via teacher forcing (clean context)}
\end{eqbox}

\vspace{0.5em}
\hi{How it works:}
\begin{itemize}
    \item Student receives \hi{clean} context frames
          (teacher forcing)
    \item Learns to map noisy inputs to clean data
          in fewer steps
    \item Provides a stable initialization
          before self-generated training
\end{itemize}
\end{frame}

% --- Slide 19b: Self Forcing Distillation ---
\begin{frame}{Self Forcing: Distillation Stage}
\hi{Stage 2: Distillation} --- student trains on its own outputs
\citep{huang2025self}

\vspace{0.3em}
\begin{eqbox}
\begin{equation*}
\nabla\mathcal{L}_{\text{distill}}
  = -\mathbb{E}_{z,t}\Big[
    (s_{\text{real}}(x_{t},t)
    - s_{\text{fake}}(x_{t},t))
    \frac{dG_{\text{student}}}{d\theta}\Big]
\end{equation*}
{\small KL divergence between teacher and student
distributions \citep{yin2024one}}
\end{eqbox}

\vspace{0.2em}
{\small $s_{\text{real}}$: score from real (teacher-generated) data
\quad $s_{\text{fake}}$: score from fake (student-generated) data}

\vspace{0.3em}
\hi{Extended rollout:} Student generates $N'>N$ frames,
loss on last $N$ --- reduces long-horizon drift.
\end{frame}

% --- Slide 28: Code LAM ---
\begin{frame}{Code: Latent Action Model}
\begin{methodbox}[Implementation Details (Not in Paper)]
\begin{itemize}
    \item Class \texttt{LatentActionModel} wraps
          \hi{asymmetric} encoder--decoder
    \item Encoder: \texttt{SpatioTemporalTransformer}
          (joint space-time attention across frames)
    \item Decoder: \texttt{SpatioTransformer}
          (spatial-only --- no temporal mixing)
    \item Action prompt initialized as:\\
          {\small \texttt{nn.Parameter(torch.empty(1, 1, 1, patch\_token\_dim))}}
\end{itemize}
\end{methodbox}

\vspace{0.3em}
\hi{Key asymmetry:} The encoder sees \hi{both} frames
jointly (spatiotemporal), while the decoder reconstructs
from action + \hi{single} frame (spatial only).
This forces all temporal information through the
32-dim bottleneck.
\end{frame}

% --- Slide 29: Code DiT ---
\begin{frame}{Code: DiT Action Injection}
\begin{methodbox}[Concrete Code Patterns]
\begin{itemize}
    \item Temporal rearrange:\\
          {\small \texttt{rearrange(action, "b 1 (t d) -> b t d", t=...)}}
    \item \hi{Two separate MLPs} for action embedding:\\
          {\small \texttt{action\_embedder\_B\_D}: global timestep path\\
          \texttt{action\_embedder\_B\_3D}: per-frame spatial path}
    \item Each MLP last layer \hi{zero-initialized}
    \item Embeddings added to timestep \emph{before} AdaLN
\end{itemize}
\end{methodbox}

\vspace{0.3em}
\begin{softbox}
\higold{My Take:} Two separate action MLPs ---
not one --- feed into different pathways.
This dual-path design is more sophisticated
than the paper describes and explains why
chunked injection has such a large ablation effect.
\end{softbox}
\end{frame}

% --- Slide 30: Code Pipeline ---
\begin{frame}{Code: Data Pipeline}
\begin{methodbox}[Three Distinct Loaders]
\begin{itemize}
    \item \hi{MANO loader:} For EgoDex/In-lab
          with hand pose annotations
    \item \hi{VideoOnly loader:} For DreamDojo-HV
          (latent actions extracted online)
    \item \hi{LeRobot loader:} For target robot data
          with ground-truth joint actions
\end{itemize}
\end{methodbox}

\vspace{0.3em}
\hi{Dual-resolution processing:}\\
{\small LAM operates at $320\times240$ (efficiency),
while the world model trains at $640\times480$ (quality).
Separate tokenizer passes for each resolution.}
\end{frame}

% --- Slide 31: Code Relative Actions ---
\begin{frame}{Code: Relative Action Computation}
\begin{methodbox}[SE(3) Math for Relative Actions]
\begin{itemize}
    \item Translation: body-frame deltas,
          not naive world-frame subtraction
    \item Rotation: proper rotation matrix computation
          for relative orientation
    \item Scale factor: $\times$20 applied
          to normalize action magnitudes
    \item Rebaseline every 4 timesteps
          (matching tokenizer compression)
\end{itemize}
\end{methodbox}

\vspace{0.5em}
\begin{softbox}
\higold{My Take:} The relative action computation uses
proper SE(3) math --- translation deltas in body frame,
not naive subtraction. The \hi{20$\times$ scaling factor}
is a practical detail the paper omits entirely.
These ``boring'' engineering choices are
what make the model actually work.
\end{softbox}
\end{frame}

% --- Slide 33: DreamDojo vs DreamZero Paradigms ---
\begin{frame}{DreamDojo vs.\ DreamZero: Two Paradigms}
\begin{columns}[T]
\column{0.48\textwidth}
\hi{DreamZero (WAM = Policy):}
\begin{itemize}
    \item \hi{Inverse} dynamics
    \item Video + action \emph{jointly} generated
    \item Zero-shot policy transfer
    \item Trained on robot data only
    \item 500 hours robot teleoperation
\end{itemize}

\column{0.48\textwidth}
\hi{DreamDojo (World Model):}
\begin{itemize}
    \item \hi{Forward} dynamics
    \item Action $\rightarrow$ video prediction
    \item Simulation, planning, teleoperation
    \item Pretrained on \higold{44K hours human video}
    \item Adapted to robots via post-training
\end{itemize}
\end{columns}

\vspace{0.3em}
\begin{keybox}[Complementary, Not Competing]
DreamZero asks ``what action produces this future?''\\
DreamDojo asks ``what future does this action produce?''
\end{keybox}
\end{frame}

% --- Slide 35: Key Takeaways ---
\begin{frame}{Key Takeaways}
\begin{enumerate}
    \item \hi{Human video pretraining works}

    \item \hi{Latent actions bridge embodiments}

          infinitely scalable
    \item \hi{Chunked + relative actions}

          general-purpose simulators}

\end{enumerate}
