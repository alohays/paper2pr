\begin{frame}{Why Not Just Use VLMs for Robots?}
\begin{itemize}
    \item VLMs are \negative{language-first} ---
          vision is a ``second-class citizen''
    \item Most VLM parameters encode \emph{knowledge},
          not \emph{physics}
    \item Static image--text pretraining
          $\neq$ spatiotemporal understanding
\end{itemize}

\vspace{0.5em}
\begin{softbox}
\higold{Commentary:} ``Language is a bottleneck, a scaffold,
not a foundation'' for physical intelligence.
\end{softbox}

% --- Slide 6: WAM Big Picture ---
\begin{frame}{World Action Models: The Big Picture}
\hi{WAM} = Video Prediction $\times$ Inverse Dynamics Model

\vspace{0.5em}
\begin{eqbox}
\begin{equation*}
\underbrace{P(\mathbf{O}, \mathbf{A} \mid \text{ctx})}_{\text{DreamZero}}
  = \underbrace{P(\mathbf{O} \mid \text{ctx})}_{\text{video pred.}}
  \times \underbrace{P(\mathbf{A} \mid \mathbf{O},
  \text{ctx})}_{\text{IDM}}
\end{equation*}
\end{eqbox}

\vspace{0.3em}
{\small\hi{IDM (Inverse Dynamics Model):} observes a sequence of states
and infers what actions caused those transitions ---
watching video and reverse-engineering motor commands.}

\vspace{0.5em}
\begin{itemize}
    \item Instead of two separate models
          $\rightarrow$ \hi{single end-to-end model}
    \item \hi{DreamZero:} 14B autoregressive WAM built on
          \higold{Wan2.1} video backbone \citep{wan2025wan}
\end{itemize}
\end{frame}

% --- Slide 9: Why AR? ---
\begin{frame}{Why Autoregressive? AR vs.\ Bidirectional}
\begin{columns}[T]
\column{0.48\textwidth}
\hi{Bidirectional problems:}
\begin{itemize}
    \item Fixed-length sequences
          $\rightarrow$ \negative{video subsampling}
    \item Distorts native FPS
          $\rightarrow$ breaks alignment
    \item Especially harmful mid-task
\end{itemize}

\vspace{0.3em}
\includegraphics[width=\textwidth,
  height=0.28\textheight,keepaspectratio]{%
  ../Figures/DreamZero/bidiraug.pdf}

\column{0.48\textwidth}
\hi{Autoregressive advantages:}
\begin{itemize}
    \item Preserves \positive{native frame rate}
    \item \positive{KV-cache} for 3--4$\times$ speedup
    \item Natural modality alignment
    \item Variable-length training
\end{itemize}
\end{columns}
\end{frame}

% --- Slide 9b: AR — Commentary ---
\begin{frame}{AR vs.\ Bidirectional: Commentary}
\begin{softbox}
\higold{Commentary:} AR wins on both quality AND speed ---
a rare ``have your cake and eat it too'' situation.
\end{softbox}

\vspace{0.5em}
\begin{itemize}
    \item \hi{Speed:} KV-cache gives 3--4$\times$ speedup ---
          bidirectional cannot cache
    \item \hi{Quality:} native frame rate preserves the temporal
          signal that the IDM needs
\end{itemize}
\end{frame}

% --- Slide 10: Chunk-wise Generation ---
\begin{frame}{Chunk-wise Generation}
\begin{methodbox}[Chunking Strategy]
\begin{itemize}
    \item Video: \hi{$K=2$ latent frames} per chunk
    \item Action horizon: \hi{$H=48$} (AgiBot, 30Hz)
          or \hi{$H=24$} (DROID, 15Hz)
    \item Each chunk $\approx$ \higold{1.6 seconds}
    \item Max 4 chunks $\rightarrow$ \higold{6.6 seconds} context
\end{itemize}
\end{methodbox}

\vspace{0.5em}
\hi{Key design choices:}
\begin{itemize}
    \item Variable-length training (like LLM token sequences)
    \item Video at 5 FPS, actions at 30Hz (AgiBot)
    \item 33 raw frames $\rightarrow$ 8 latent frames ($4\times2$)
\end{itemize}
\end{frame}

% --- Slide 11: Flow Matching Intuition ---
\begin{frame}{Flow Matching: Intuition}
\begin{keybox}[What is Flow Matching?]
Learn a \hi{velocity field} that transports noise to data
along straight paths --- simpler and faster than
standard diffusion.
\end{keybox}

\vspace{0.3em}
{\small\emph{Analogy:} Think of a GPS that plots a
straight-line route from ``lost'' (noise) to ``home'' (data).
Flow matching learns that route; standard diffusion
takes a random walk.}

\vspace{0.5em}
\begin{itemize}
    \item At $t=0$: pure noise. At $t=1$: clean data.
    \item The model predicts the \hi{velocity}
          pointing from noise to data.
    \item \hi{Shared timestep} $t_k$ for video and action
          $\rightarrow$ aligned convergence.
\end{itemize}
\end{frame}

% --- Slide 11b: Flow Matching Equations ---
\begin{frame}{Flow Matching: Equations}
\begin{eqbox}
\hi{Noisy interpolation:}
\begin{equation*}
\mathbf{z}_{t_k}^k = t_k \mathbf{z}_1^k
  + (1-t_k)\mathbf{z}_0^k, \quad
\mathbf{a}_{t_k}^k = t_k \mathbf{a}_1^k
  + (1-t_k)\mathbf{a}_0^k
\end{equation*}
where $\mathbf{z}_0^k, \mathbf{a}_0^k
  \sim \mathcal{N}(\mathbf{0}, \mathbf{I})$
and $t_k \in [0,1]$.
\end{eqbox}

\vspace{0.5em}
\begin{eqbox}
\hi{Velocity prediction loss:}
\begin{equation*}
\mathcal{L}(\theta) = \mathbb{E}\Bigg[
  \frac{1}{K}\sum_{k=1}^{K} w(t_k)
  \Big\lVert \mathbf{u}_\theta(\cdots)
  - \mathbf{v}^k \Big\rVert^2\Bigg]
\end{equation*}
\end{eqbox}
\end{frame}

% --- Slide 11c: Flow Matching Notation ---
\begin{frame}{Flow Matching: Notation Guide}
\hi{Key Symbols:}
\begin{itemize}
    \item $\mathbf{u}_\theta$: the DiT model
          (predicts velocity)
    \item $\mathcal{C}_k$: clean context
          from previous chunks
    \item $\mathbf{c}$: language instruction
    \item $\mathbf{q}_k$: proprioceptive state
    \item $w(t_k)$: predefined weight function
\end{itemize}

\vspace{0.5em}
\hi{Teacher forcing:} denoise noisy current chunk
conditioned on \positive{clean} previous chunks
(like LLM training with ground-truth tokens).
\end{frame}

% --- Slide 13: Ground-Truth Injection ---
\begin{frame}{Inference: Ground-Truth Injection}
\begin{keybox}[The Elegant Trick]
\begin{enumerate}
    \item Denoise one chunk at a time (autoregressive)
    \item Execute action chunk on robot
          $\rightarrow$ get \hi{real observation}
    \item \higold{Replace predicted frame with ground-truth}
          in KV cache
    \item This \positive{eliminates error accumulation!}
\end{enumerate}
\end{keybox}

\vspace{0.5em}
\hi{Why this is unique to WAMs:}
\begin{itemize}
    \item Pure video generation: errors compound
    \item WAMs operate in \positive{closed-loop} ---
          reality corrects the model
\end{itemize}
\end{frame}

% --- Slide 13b: GT Injection — Commentary ---
\begin{frame}{Ground-Truth Injection: Commentary}
\begin{softbox}
\higold{Commentary:} WAMs \hi{self-correct via reality}.
No other paradigm gets this for free.
\end{softbox}

\vspace{0.5em}
\begin{itemize}
    \item Closes the sim-to-real gap \emph{by design},
          not by domain randomization
    \item Compare to pure video generation
          (e.g.\ Sora) where errors compound indefinitely
\end{itemize}
\end{frame}

% --- Slide 14: DreamZero-Flash ---
\begin{frame}{DreamZero-Flash: Faster Inference}
\hi{Problem:} 4 denoising steps = 350ms
(too slow for some applications)

\vspace{0.5em}
\hi{Solution:} \higold{Decoupled noise scheduling}
\begin{itemize}
    \item Video: $t_{\text{vid}} = 1 - \eta$,
          $\eta \sim \text{Beta}(7,1)$\\
          $\rightarrow$ biased toward \negative{high-noise}
          (mean $t \approx 0.125$)
    \item Action: $t_{\text{act}} \sim \text{Uniform}[0,1]$
          $\rightarrow$ standard
\end{itemize}

\vspace{0.3em}
Trains actions to denoise from
\emph{predominantly noisy} visual context.
\end{frame}

% --- Slide 16: Data Philosophy ---
\begin{frame}{Data Philosophy: Diversity Over Repetition}
\begin{resultbox}[Counterintuitive Finding]
Same 500 hours of data:\\
\hi{Diverse} (22 envs, long-tail skills):
  \positive{50\% task progress}\\
\hi{Repetitive} (70 tasks, many reps):
  \negative{33\% task progress}\\
$\rightarrow$ \positive{+51\% relative improvement} from diversity
\end{resultbox}

\vspace{0.5em}
\hi{Why?}
\begin{itemize}
    \item Video prediction inherited from pretraining
    \item Robust IDM requires
          \positive{diverse state-action correspondences}
    \item Repetitive data $\rightarrow$ narrow IDM
\end{itemize}
\end{frame}

than robot data.
\end{softbox}

          how-to and activity video
    \item The bottleneck shifts from ``collect robot data''
          to ``build a better IDM''
\end{itemize}

% --- Slide 27: VLA vs WAM ---
\begin{frame}{Why VLAs Fail While WAMs Succeed}
\begin{columns}[T]
\column{0.48\textwidth}
\hi{VLAs:}
\begin{itemize}
    \item \negative{Static images}
          $\rightarrow$ no temporal dynamics
    \item Scaling does not help:
          \negative{0\% at 5B and 14B}
    \item Repetitive data + static priors
          = \negative{overfitting}
\end{itemize}

\column{0.48\textwidth}
\hi{WAMs:}
\begin{itemize}
    \item \positive{Video}
          $\rightarrow$ rich temporal dynamics
    \item Scaling improves video quality
          $\rightarrow$ \positive{better actions}
    \item Diverse data + video priors
          = \positive{generalization}
\end{itemize}
\end{columns}
\end{frame}

% --- Slide 27b: VLA vs WAM — Commentary ---
\begin{frame}{VLA vs.\ WAM: Commentary}
\begin{softbox}
\higold{Commentary:} A \hi{fundamental architectural
difference}, not data/compute. VLMs encode semantics;
video models encode physics. For robotics, physics wins.
\end{softbox}

\vspace{0.5em}
\begin{itemize}
    \item This echoes the ``bitter lesson'':
          the right \emph{representation}
          (video) matters more than scale
    \item Expect hybrid approaches that combine
          VLM semantics with WAM physics
\end{itemize}
\end{frame}

    \item IDM faithfully extracts actions from wrong video
    \item Robot executes incorrect motion

% --- Slide 32: Limitations ---
\begin{frame}{Limitations}
\hi{From the paper:}
\begin{enumerate}
    \item No established \negative{scaling laws} for WAMs
    \item Context window limited to \negative{6.6 seconds}
    \item Still \negative{7Hz} vs.\ 20Hz+ for VLAs
    \item Inherits behavior cloning limitations
          on \negative{sub-centimeter precision}
    \item Separate training per embodiment
\end{enumerate}
\end{frame}

% --- Slide 32b: Computational Requirements ---
\begin{frame}{Limitations: Computational Requirements}
\hi{Computational requirements:}
\begin{itemize}
    \item 2$\times$GB200 GPUs for real-time control
    \item Not yet feasible on consumer hardware
    \item 14B parameters = significant VRAM
\end{itemize}

\vspace{0.5em}
\hi{Long-horizon:}
\begin{itemize}
    \item 6.6s context $\rightarrow$ System 1 only
    \item Needs System 2 planner for extended tasks
\end{itemize}
\end{frame}

% --- Slide 33: Unsolved IDM ---
\begin{frame}{The Unsolved Inverse Dynamics Problem}
\hi{Adaptation is not free.}

\vspace{0.5em}
\hi{The credit card analogy:}
\begin{itemize}
    \item Dexterous hands: slide + pinch
    \item Grippers: push from edge
    \item \emph{Same task, different motions}
\end{itemize}

\vspace{0.5em}
\hi{IDM complexity scales with DOF:}
\begin{itemize}
    \item Simple grippers: relatively easy
    \item Dexterous hands: much harder
    \item Full humanoids: \negative{hardest}
\end{itemize}
\end{frame}

% --- Slide 34: Hardware vs Algorithm ---
\begin{frame}{The Hardware vs.\ Algorithm Race}

\column{0.48\textwidth}
\hi{Hardware Path:}\\
If humanoid actuators mature quickly
$\rightarrow$ embodiment matching removes transfer loss

\vspace{0.5em}
\hi{Algorithm Path:}\\
If embodiment adaptation improves faster
$\rightarrow$ intermediate morphologies remain viable

We are back to the age of research ---
fundamentals matter again.

Both paths converge on the same data:\\
\higold{human video}
\end{columns}
\end{frame}
